# vogue v0.2 — Plan 5: Lead-lag estimation

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Estimate the temporal lead between two year-series — e.g. does publication fashion (`openalex:raw`) precede funding (`gepris:coded_trope`)? — via detrended cross-correlation, and expose it as a `leadlag` CLI command. This is the analysis that turns "the acceleration thesis" from a slogan into a number.

**Architecture:** `analysis.cross_correlate(a, b, max_lag, method)` aligns two `{year: count}` series, **detrends** each (critical: both series rise with corpus growth, so raw cross-correlation just finds the shared trend — detrending isolates the *shape*), and reports the lag maximizing Pearson correlation. A `leadlag` CLI command builds the overlay and runs it on two chosen series.

**Tech Stack:** pure Python (no scipy — Pearson and detrending implemented directly). Builds on Plan 3/4 (`Overlay`, `build_overlay`).

**Environment:** uv. Tests via `uv run pytest -q`, lint via `uv run ruff check vogue tests`.

Work on branch `v0.2-leadlag`. Commit after each task. Additive only — no existing public names/signatures change.

```python
    """Triangular bump peaking at `center` (height 10), zero far away."""
    return {y: max(0, 10 - abs(y - center)) for y in range(lo, hi + 1)}


def test_detrend_methods():

    assert detrend([0, 5, 10], "peak") == [0.0, 0.5, 1.0]
    assert detrend([1, 2, 3], "diff") == [1, 1]

    assert abs(sum(z)) < 1e-9 and z[0] < 0 < z[2]
    assert detrend([5, 5, 5], "zscore") == [0.0, 0.0, 0.0]  # zero variance

    assert abs(pearson([1, 2, 3], [2, 4, 6]) - 1.0) < 1e-9
    assert abs(pearson([1, 2, 3], [3, 2, 1]) + 1.0) < 1e-9
    assert pearson([1, 1, 1], [1, 2, 3]) == 0.0  # zero variance -> 0
```

```python
MIN_PAIRS = 4  # minimum aligned year-pairs to trust a correlation at a given lag


def detrend(values: list[float], method: str) -> list[float]:
    """Remove the level/trend so cross-correlation sees shape, not shared growth."""
    if method == "none":
        return list(values)
    if method == "peak":
        m = max(values) if values else 0
        return [v / m for v in values] if m else list(values)
    if method == "diff":
        return [values[i] - values[i - 1] for i in range(1, len(values))]
    if method == "zscore":
        n = len(values)
        if n == 0:
            return []
        mu = sum(values) / n
        sd = (sum((v - mu) ** 2 for v in values) / n) ** 0.5
        return [(v - mu) / sd for v in values] if sd > 0 else [0.0] * n
    raise ValueError(f"unknown detrend method: {method}")


def pearson(xs: list[float], ys: list[float]) -> float:
    n = len(xs)
    if n < 2:
        return 0.0
    mx, my = sum(xs) / n, sum(ys) / n
    cov = sum((x - mx) * (y - my) for x, y in zip(xs, ys))
    vx = sum((x - mx) ** 2 for x in xs)
    vy = sum((y - my) ** 2 for y in ys)
    if vx <= 0 or vy <= 0:
        return 0.0
    return cov / (vx * vy) ** 0.5


@dataclass
class LeadLag:
    best_lag: int           # +k => series b leads series a by k years
    best_corr: float
    profile: dict[int, float]
    method: str
    n_pairs: int            # aligned year-pairs at best_lag


def cross_correlate(a: dict[int, int], b: dict[int, int],
                    max_lag: int = 8, method: str = "zscore") -> LeadLag | None:
    """Best lag k maximizing corr(a[y], b[y-k]) over detrended, year-aligned series.
    Positive best_lag means b precedes a (b leads a) by that many years. None if too thin."""
    if not a or not b:
        return None
    a_years = range(min(a), max(a) + 1)
    b_years = set(range(min(b), max(b) + 1))
    profile: dict[int, float] = {}
    pairs_at: dict[int, int] = {}
    for k in range(-max_lag, max_lag + 1):
        xs, ys = [], []
        for y in a_years:
            if (y - k) in b_years:
                xs.append(a.get(y, 0))
                ys.append(b.get(y - k, 0))
        dx, dy = detrend(xs, method), detrend(ys, method)
        if len(dx) >= MIN_PAIRS:
            profile[k] = pearson(dx, dy)
            pairs_at[k] = len(xs)
    if not profile:
        return None
    best = max(profile, key=lambda k: profile[k])
    return LeadLag(best_lag=best, best_corr=profile[best], profile=profile,
                   method=method, n_pairs=pairs_at[best])
```

```python
@app.command()
def leadlag(study_dir: str, term: str = typer.Option(...),
            lead: str = typer.Option("openalex:raw", help="series expected to lead"),
            lag: str = typer.Option("gepris:coded_trope", help="series expected to lag"),
            method: str = typer.Option("zscore", help="detrend: zscore|diff|peak|none"),
            max_lag: int = typer.Option(8), limit: int = typer.Option(2000)):
    """Estimate how many years `lead` precedes `lag` via detrended cross-correlation."""
    study = Study.load(study_dir)
    overlay = build_overlay(study, term, limit=limit)
    a, b = overlay.series.get(lag), overlay.series.get(lead)
    if not a or not b:
        typer.echo(f"need both series present; available: {sorted(overlay.series)}")
        raise typer.Exit(1)
    res = cross_correlate(a, b, max_lag=max_lag, method=method)
    if res is None:
        typer.echo("insufficient overlapping data for a lead-lag estimate")
        raise typer.Exit(1)
    direction = "leads" if res.best_lag >= 0 else "lags"
    typer.echo(
        f"{term}: '{lead}' {direction} '{lag}' by {abs(res.best_lag)} years "
        f"(r={res.best_corr:.2f}, method={method}, n={res.n_pairs})")
```
